### 项目上线
1. npm init -y
2. npm i express -S
3. dist
4. app.js
```js
const express = require('express')

const app = express()

app.use(express.static('./dist'))

app.listen(80, () => {
    console.log('web serve running at http://127.0.0.1');
})
```
5. node .\app.js
6. http://127.0.0.1
